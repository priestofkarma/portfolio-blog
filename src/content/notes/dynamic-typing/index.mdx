### Переклад у рядок

```jsx
// To string
console.log(String(null)); // "null" - строка
console.log(String(4)); // "4" - строка

// Все, що конкатинуємо з рядком - тип даних рядок
console.log(typeof (5 + "")); // тип даних - string

const num = 5;

console.log("https://facebook.com/catalog/" + num); // https://facebook.com/catalog/5

const fontSize = 26 + "px"; // 26px
```

### Переведення в числовий тип даних

```jsx
// To number
console.log(typeof Number("4")); // number

// Унарний плюс
typeof +"5"; // number

// parseInt
typeof parseInt("15px", 10);

let answ = +prompt("Hello", "");
```

### Переклад чогось у булеве значення

```jsx
// To false
0, '', null, undefined, NaN -> FALSE

let switcher = null;

if(switcher) { // false
	console.log('Working...');
}

switcher = 1;

if(switcher) { // true
	console.log('Working...');
}

typeof(Boolean('4')); // boolean

(!!'312313') // два знаки заперечення перед чимось, перетворюють на boolean
```
