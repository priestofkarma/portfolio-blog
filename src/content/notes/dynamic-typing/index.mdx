### Перевод в строку

```jsx
// To string
console.log(String(null)); // "null" - строка
console.log(String(4)); // "4" - строка

// Все, что конкатинируем со строкой - тип данных строка
console.log(typeof (5 + "")); // тип данных - string

const num = 5;

console.log("https://vk.com/catalog/" + num); // https://vk.com/catalog/5

const fontSize = 26 + "px"; // 26px
```

### Перевод в числовой тип данных

```jsx
// To number
console.log(typeof Number("4")); // number

// Унарный плюс
typeof +"5"; // number

// parseInt
typeof parseInt("15px", 10);

let answ = +prompt("Hello", "");
```

### Перевод чего-либо в булевое значение

```jsx
// To false
0, '', null, undefined, NaN -> FALSE

let switcher = null;

if(switcher) { // false
	console.log('Working...');
}

switcher = 1;

if(switcher) { // true
	console.log('Working...');
}

typeof(Boolean('4')); // boolean

(!!'312313') // два знака отрицания перед чем либо, превращают в boolean
```
