### Властивості

```jsx
//Властивість length
const str = "test";
str.length // 4

const arr = [1, 2, 3];
arr.length // 3
```

### Методи рядків

```jsx
//Властивість length
const str = "test";

str.toUpperCase(); // TEST
str.toLowerCase(); // test

consoloe.log(str) // test (значення залишається)

// --------------------------

const fruit = "Some fruit";

fruit.indexOf("fruit"); // 5
fruit.indexOf("q"); // -1 не знайдено

// --------------------------

const logg = "Hello World";
console.log(logg.slice(6, 10)); // Worl (обрізає з 6-го символу до 10 не влучно)
console.log(logg.slice(6)); // World (обрізає весь рядок, починаючи з 6-го символу)
console.log(logg.slice(-5, -1)); // Worl (обрізає весь рядок починаючи з 5 символу, рахуючи з кінця до передостаннього символу)
console.log(logg.substring(6, 11)); // не підтримує орієнтовні значення (-6) = 0
console.log(logg.substr(4, 3)); // o W (починаючи з 4-го символу вирізати 3 наступні символи)
```

### Методи чисел

```jsx
const num = 12.2;
Math.round(num); // 12

const test = "12.2px";
parseInt(test); // 12
parseFloat(test); // 12.2
```
