### Свойства

```jsx
//Свойство length
const str = "test";
str.length // 4

const arr = [1, 2, 3];
arr.length // 3
```

### Методы строк

```jsx
//Свойство length
const str = "test";

str.toUpperCase(); // TEST
str.toLowerCase(); // test

consoloe.log(str) // test (значение остается)

// --------------------------

const fruit = "Some fruit";

fruit.indexOf("fruit"); // 5
fruit.indexOf("q"); // -1 не найдено

// --------------------------

const logg = "Hello World";
console.log(logg.slice(6, 10)); // Worl (обрезает с 6-го символа до 10 не влючительно)
console.log(logg.slice(6)); // World (обрезает всю строку начиная с 6-го символа)
console.log(logg.slice(-5, -1)); // Worl (обрезает всю строку начиная с 5 символа считая с конца до предпоследнего символа)
console.log(logg.substring(6, 11)); // не поддерживает орицательные значения (-6) = 0
console.log(logg.substr(4, 3)); // o W (начиная с 4-го символа вырезать 3 следющих символа)
```

### Методы чисел

```jsx
const num = 12.2;
Math.round(num); // 12

const test = "12.2px";
parseInt(test); // 12
parseFloat(test); // 12.2
```
