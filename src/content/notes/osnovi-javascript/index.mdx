### Оператори в JS

#### Конкатенація

```jsx
// Рядок + рядок
'arr' + ' - object' // arr - object

// Число + рядок
4 + ' - object' // 4 - object
// Якщо рядок перетворити на число унарним плюсом
4 + +' - object' // NaN
4 + +'5' // 9
```

#### Інкремент та декремент

```jsx

let incr = 10,
	decr = 10;

// Постфіксна форма
incr++;
decr--;

console.log(incr, decr); // 11, 9

// Префіксна форма
++incr;
--decr;

console.log(incr, decr); // 11, 9

/* Якщо використати відразу,
  * Постфіксна форма спочатку повертає
  * Старе значення (10), а вже після нього
  * збільшує або зменшує
*/

console.log(incr++, decr--); // 10, 10
```

#### Оператор залишок

```jsx
// (%) повертає залишок від розподілу
5%2 // 1
```

#### Оператор заперечення

```jsx
// Оператор заперечення !
isChecked || !isClose // false
```

#### Оператор рівності

```jsx
// Оператор рівності "==" и "==="
2*4 == 8 // true
2*4 == "8" // true
2*4 === "8" // false (суворе порівняння, порівнює за типом даних)

// Оператор рівності із запереченням "!=" та суворим запереченням "!=="
2 + 2 * 2 != 8 // true
2 + 2 * 2 != '6' // false
2 + 2 * 2 !== '6' // true
```

#### Логічні оператори

```jsx
// Логічний оператор І &&
const isChecked = true,
			isClose = true;

isChecked && isClose // true

// Логічний оператор АБО || 
const isChecked = true,
	  isClose = false;

isChecked || isClose // true

// Оператор із запереченням
isChecked && !isClose // true
```
