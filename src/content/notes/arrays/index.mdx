### Методи масивів

```jsx
const arr = [2, 13, 26, 8, 10];

// Видалити останній елемент масиву
arr.pop();

// Додати елемент до кінця масиву
arr.push(10);

// З рядка зробити масив
const str = "qqq, www, eee, rrr, ttt";
const products = str.split(", ");

console.log(products); // ["qqq", "www", "eee", "rrr", "ttt"]

// Об'єднати елементи масиву в рядок
console.log(products.join('; ')); "qqq; www; eee; rrr; ttt;"

// Сортування об'єкту
// Цей метод сортує рядки (за алфавітом)
products.sort();

arr.sort(); // [10, 13, 2, 26, 8]

function compareNum(a, b) {
	return a - b;
}

arr.sort(); // [2, 8, 10, 13, 26]

```

### Особливість якості length

```jsx
const arr = [1, 2, 3, 6, 8];

arr[99] = 0;

// Властивість length складається з останнього індексу + 1
console.log(arr.length); // 100

console.log(arr); // [1, 2, 3, 6, 8, <94 empty items>, 0]
```

### Перебір масивів

У переборі forEach не можна використовувати break і continue

```jsx
// Через цикл for
for (let i = 0; i < arr.length; i++) {
	console.log(arr[i]);
}

// Через цикл for of
for (let value of arr) {
	console.log(value);
}

// Через forEach
arr.forEach(function(item, index, array) {
	console.log(`${index}: ${item} всередині масиву ${array]`);
});
```
