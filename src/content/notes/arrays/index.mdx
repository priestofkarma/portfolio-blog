### Методы массивов

```jsx
const arr = [2, 13, 26, 8, 10];

// Удалить последний элемент массива
arr.pop();

// Добавить элемент в конец массива
arr.push(10);

// Из строки сдлеать массив
const str = "qqq, www, eee, rrr, ttt";
const products = str.split(", ");

console.log(products); // ["qqq", "www", "eee", "rrr", "ttt"]

// Объеденить элементы массива в строку
console.log(products.join('; ')); "qqq; www; eee; rrr; ttt;"

// Сортировка объекта
// Этот метод сортирует строки (по алфавиту)
products.sort();

arr.sort(); // [10, 13, 2, 26, 8]

function compareNum(a, b) {
	return a - b;
}

arr.sort(); // [2, 8, 10, 13, 26]

```

### Особенность свойства length

```jsx
const arr = [1, 2, 3, 6, 8];

arr[99] = 0;
// Свойство length состоит из последнего индекса + 1
console.log(arr.length); // 100

console.log(arr); // [1, 2, 3, 6, 8, <94 empty items>, 0]
```

### Перебор массивов

В переборе forEach нельзя использовать break и continue

```jsx
// Через цикл for
for (let i = 0; i < arr.length; i++) {
	console.log(arr[i]);
}

// Через цикл for of
for (let value of arr) {
	console.log(value);
}

// Через forEach
arr.forEach(function(item, index, array) {
	console.log(`${index}: ${item} внутри массива ${array]`);
});
```
